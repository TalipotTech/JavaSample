4.1

Introduction to Arrays

Understanding why we need arrays and their basic concepts

üì¶ What is an Array?

An array is a container object that holds a fixed number of values of a single type. Think of it as a row of boxes, where each box can store one item of the same type.

Instead of declaring multiple variables for related data, arrays let you store multiple values in a single variable with indexed access.

üí° Key Point

Array indices always start at 0 (zero-indexed). So the first element is at index 0, the second at index 1, and so on. For an array of length 5, valid indices are 0 through 4.

üìê

Fixed Size

Once created, an array's size cannot change. You must know the size beforehand or use dynamic collections like ArrayList.

üéØ

Same Type

All elements must be the same data type. An int[] can only hold integers, a String[] only strings.

‚ö°

Fast Access

Accessing any element by index takes constant time O(1) ‚Äì extremely fast regardless of array size.

üî¢

Zero-Indexed

Indices start at 0. The first element is array[0], the last is array[length-1].

‚ùì Why Use Arrays?

Consider storing test scores for 100 students. Without arrays, you'd need 100 separate variables!

üîç Searching Arrays

Two common search algorithms: Linear Search (for unsorted) and Binary Search (for sorted arrays).

ArraySearching.java
Java 
import java.util.Arrays;

public class ArraySearching {

    // Linear Search - works on any array
    public static int linearSearch(int[] arr, int target) {
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == target) {
                return i;  // Found! Return index
            }
        }
        return -1;  // Not found
    }
    
    // Binary Search - requires SORTED array
    public static int binarySearch(int[] arr, int target) {
        int left = 0;
        int right = arr.length - 1;
        
        while (left <= right) {
            int mid = left + (right - left) / 2;
            
            if (arr[mid] == target) {
                return mid;  // Found!
            } else if (arr[mid] < target) {
                left = mid + 1;  // Search right half
            } else {
                right = mid - 1;  // Search left half
            }
        }
        return -1;  // Not found
    }

    public static void main(String[] args) {
        int[] unsorted = {45, 12, 89, 33, 67, 8, 91};
        int[] sorted = {8, 12, 33, 45, 67, 89, 91};
        
        // Linear Search
        System.out.println("=== Linear Search ===");
        int result1 = linearSearch(unsorted, 67);
        System.out.println("Search for 67: index = " + result1);
        
        int result2 = linearSearch(unsorted, 100);
        System.out.println("Search for 100: index = " + result2 + " (not found)");
        
        // Binary Search
        System.out.println("\n=== Binary Search ===");
        int result3 = binarySearch(sorted, 67);
        System.out.println("Search for 67: index = " + result3);
        
        // Using Arrays.binarySearch() - built-in method
        System.out.println("\n=== Arrays.binarySearch() ===");
        int result4 = Arrays.binarySearch(sorted, 33);
        System.out.println("Search for 33: index = " + result4);
    }
}

üí° Performance Comparison

Linear Search: O(n) - checks each element; works on any array.
Binary Search: O(log n) - much faster, but requires a sorted array. For 1 million elements: Linear needs up to 1,000,000 comparisons; Binary needs only ~20!

üìà Sorting Arrays

Use the Arrays.sort() method for efficient sorting, or implement your own algorithms:

ArraySorting.java
Java 
import java.util.Arrays;

public class ArraySorting {

    // Bubble Sort - simple but slow O(n¬≤)
    public static void bubbleSort(int[] arr) {
        int n = arr.length;
        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < n - i - 1; j++) {
                if (arr[j] > arr[j + 1]) {
                    // Swap
                    int temp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = temp;
                }
            }
        }
    }
    
    // Selection Sort - finds minimum and swaps
    public static void selectionSort(int[] arr) {
        int n = arr.length;
        for (int i = 0; i < n - 1; i++) {
            int minIdx = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIdx]) {
                    minIdx = j;
                }
            }
            // Swap
            int temp = arr[minIdx];
            arr[minIdx] = arr[i];
            arr[i] = temp;
        }
    }

    public static void main(String[] args) {
        // Using Arrays.sort() - the recommended way
        int[] numbers = {64, 34, 25, 12, 22, 11, 90};
        
        System.out.println("Original: " + Arrays.toString(numbers));
        
        Arrays.sort(numbers);
        System.out.println("Sorted:   " + Arrays.toString(numbers));
        
        // Sorting strings alphabetically
        String[] names = {"Zara", "Alice", "Mike", "Bob"};
        System.out.println("\nNames before: " + Arrays.toString(names));
        Arrays.sort(names);
        System.out.println("Names after:  " + Arrays.toString(names));
        
        // Sorting a portion of array
        int[] partial = {5, 2, 8, 1, 9, 3, 7};
        System.out.println("\nBefore partial sort: " + Arrays.toString(partial));
        Arrays.sort(partial, 1, 5);  // Sort indices 1-4
        System.out.println("After partial sort:  " + Arrays.toString(partial));
        
        // Testing custom bubble sort
        int[] bubbleTest = {64, 34, 25, 12, 22};
        System.out.println("\n=== Bubble Sort ===");
        System.out.println("Before: " + Arrays.toString(bubbleTest));
        bubbleSort(bubbleTest);
        System.out.println("After:  " + Arrays.toString(bubbleTest));
    }
}

public class MatrixOperations {

    // Print matrix nicely
    public static void printMatrix(int[][] matrix, String name) {
        System.out.println(name + ":");
        for (int[] row : matrix) {
            System.out.print("  ‚îÇ ");
            for (int val : row) {
                System.out.printf("%4d ", val);
            }
            System.out.println("‚îÇ");
        }
    }
    
    // Matrix Addition
    public static int[][] add(int[][] a, int[][] b) {
        int rows = a.length;
        int cols = a[0].length;
        int[][] result = new int[rows][cols];
        
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                result[i][j] = a[i][j] + b[i][j];
            }
        }
        return result;
    }
    
    // Matrix Transpose
    public static int[][] transpose(int[][] matrix) {
        int rows = matrix.length;
        int cols = matrix[0].length;
        int[][] result = new int[cols][rows];
        
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                result[j][i] = matrix[i][j];
            }
        }
        return result;
    }
    
    // Sum of all elements
    public static int sumAll(int[][] matrix) {
        int sum = 0;
        for (int[] row : matrix) {
            for (int val : row) {
                sum += val;
            }
        }
        return sum;
    }
    
    // Find maximum element
    public static int findMax(int[][] matrix) {
        int max = matrix[0][0];
        for (int[] row : matrix) {
            for (int val : row) {
                if (val > max) max = val;
            }
        }
        return max;
    }

    public static void main(String[] args) {
        int[][] matrixA = {
            {1, 2, 3},
            {4, 5, 6}
        };
        
        int[][] matrixB = {
            {7, 8, 9},
            {10, 11, 12}
        };
        
        printMatrix(matrixA, "Matrix A");
        System.out.println();
        printMatrix(matrixB, "Matrix B");
        
        // Addition
        System.out.println("\n=== A + B ===");
        int[][] sum = add(matrixA, matrixB);
        printMatrix(sum, "Sum");
        
        // Transpose
        System.out.println("\n=== Transpose of A ===");
        int[][] transposed = transpose(matrixA);
        printMatrix(transposed, "A^T");
        
        // Statistics
        System.out.println("\n=== Statistics ===");
        System.out.println("Sum of A: " + sumAll(matrixA));
        System.out.println("Max in B: " + findMax(matrixB));
    }
}

import java.util.Scanner;

public class TicTacToe {
    private static char[][] board = new char[3][3];
    private static char currentPlayer = 'X';
    
    // Initialize board
    public static void initializeBoard() {
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                board[i][j] = '-';
            }
        }
    }
    
    // Print the board
    public static void printBoard() {
        System.out.println("\n    1   2   3");
        System.out.println("  ‚îå‚îÄ‚îÄ‚îÄ‚î¨‚îÄ‚îÄ‚îÄ‚î¨‚îÄ‚îÄ‚îÄ‚îê");
        for (int i = 0; i < 3; i++) {
            System.out.print((i + 1) + " ‚îÇ");
            for (int j = 0; j < 3; j++) {
                char symbol = board[i][j];
                if (symbol == 'X') {
                    System.out.print(" X ‚îÇ");
                } else if (symbol == 'O') {
                    System.out.print(" O ‚îÇ");
                } else {
                    System.out.print("   ‚îÇ");
                }

    }
    
    // Make a move
    public static boolean makeMove(int row, int col) {
        if (row < 0 || row > 2 || col < 0 || col > 2) {
            return false;
        }
        if (board[row][col] != '-') {
            return false;
        }
        board[row][col] = currentPlayer;
        return true;
    }
    
    // Check for winner
    public static boolean checkWin() {
        // Check rows and columns
        for (int i = 0; i < 3; i++) {
            if (board[i][0] != '-' && board[i][0] == board[i][1] && board[i][1] == board[i][2]) {
                return true;
            }
            if (board[0][i] != '-' && board[0][i] == board[1][i] && board[1][i] == board[2][i]) {
                return true;
            }
        }
        // Check diagonals
        if (board[0][0] != '-' && board[0][0] == board[1][1] && board[1][1] == board[2][2]) {
            return true;
        }
        if (board[0][2] != '-' && board[0][2] == board[1][1] && board[1][1] == board[2][0]) {
            return true;
        }
        return false;
    }
    
    // Check for draw
    public static boolean isBoardFull() {
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                if (board[i][j] == '-') {
                    return false;
                }
            }
        }
        return true;
    }
    
    // Switch player
    public static void switchPlayer() {
        currentPlayer = (currentPlayer == 'X') ? 'O' : 'X';
    }
